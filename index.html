from some_vector_db import VectorDB
from some_llm_provider import LargeLanguageModel

class SimpleRAG:
    def __init__(self, documents):
        self.db = VectorDB()
        self.db.add_documents(documents)
        self.llm = LargeLanguageModel()

    def query(self, question):
        retrieved_docs = self.db.search(question, top_k=3)
        context = "\\n".join(doc.content for doc in retrieved_docs)
        prompt = f"Based on the following context, answer the question.\\n\\nContext:\\n{context}\\n\\nQuestion: {question}"
        answer = self.llm.generate(prompt)
        return answer
